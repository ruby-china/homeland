def destroy

  @reply.destroy
  render json: { ok: 1 }
end

def update
  raise AccessDenied unless can?(:update, @reply)

  requires! :body

  @reply.body = params[:body]
  @reply.save!
  render json: @reply, serializer: ReplyDetailSerializer
end
